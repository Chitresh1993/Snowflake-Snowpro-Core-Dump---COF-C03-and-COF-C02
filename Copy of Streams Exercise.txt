select  n_nationkey,n_name,n_regionkey,n_comment,country_code,update_timestamp 
from nation_stream where METADATA$ACTION='INSERT' AND METADATA$ISUPDATE='TRUE';

MERGE INTO NATION_SS  NSS
USING nation_stream NTC 
ON (NSS.N_NATIONKEY = NTC.N_NATIONKEY) 
WHEN MATCHED AND (NTC.METADATA$ACTION='INSERT' AND NTC.METADATA$ISUPDATE='TRUE') THEN
UPDATE SET NSS.n_nationkey = NTC.N_NATIONKEY,NSS.n_name = NTC.n_name,NSS.n_regionkey = NTC.n_regionkey,NSS.n_comment = NTC.n_comment,
NSS.country_code = NTC.country_code,NSS.update_timestamp =NTC.update_timestamp
WHEN MATCHED AND (NTC.METADATA$ACTION='DELETE' AND NTC.METADATA$ISUPDATE='FALSE') THEN
UPDATE SET NSS.IS_DELETED = 'Y'
WHEN NOT MATCHED THEN
INSERT(n_nationkey,n_name,n_regionkey,n_comment,country_code,update_timestamp) 
VALUES (NTC.n_nationkey,NTC.n_name,NTC.n_regionkey,NTC.n_comment,NTC.country_code,NTC.update_timestamp);
