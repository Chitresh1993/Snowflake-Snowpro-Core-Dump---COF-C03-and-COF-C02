/* 

Order table Bronze and insert data query 

*/


    
CREATE OR REPLACE TABLE global_superstore.silver_superstore.S_order (
        ORDER_SK NUMBER AUTOINCREMENT START = 1000 INCREMENT = 1,
        ROW_ID STRING,
        ORDER_ID STRING,
        ORDER_DATE DATE,
        SHIP_DATE DATE,
        SHIP_MODE STRING,
        CUSTOMER_KEY NUMBER,
        -- FK to SCD2 dim customer
        COUNTRY_KEY NUMBER,
        -- FK to SCD2 dim country
        PRODUCT_KEY NUMBER,
        -- FK to SCD2 dim product
        SALES FLOAT,
        QUANTITY INT,
        DISCOUNT FLOAT,
        PROFIT FLOAT,
        _FIVETRAN_SYNCED TIMESTAMP_TZ(9),
        _sf_ingestion_timestamp TIMESTAMP_NTZ,
        Active_Flag BOOLEAN DEFAULT TRUE,
        -- Flags the current active version
        _Table_LOAD_DATETIME TIMESTAMP_NTZ DEFAULT CURRENT_TIMESTAMP
    ) CLUSTER BY (ORDER_DATE, ORDER_ID)
    DATA_RETENTION_TIME_IN_DAYS = 14 
    COMMENT = 'Order Fact Table populated from Bronze layer with surrogate key and update timestamp';

    
    /* 
    
    bulk insert data query 
    
    */

    
Insert into  global_superstore.Silver_superstore.S_ORDER (
        ROW_ID,
        ORDER_ID,
        ORDER_DATE,
        SHIP_DATE,
        SHIP_MODE,
        CUSTOMER_KEY,
        COUNTRY_KEY,
        PRODUCT_KEY,
        SALES,
        QUANTITY,
        DISCOUNT,
        PROFIT,
        _FIVETRAN_SYNCED,
        _SF_INGESTION_TIMESTAMP
    ) 
    (
        select
            distinct ROW_ID,
            ORDER_ID,
            ORDER_DATE,
            SHIP_DATE,
            SHIP_MODE,
            ABS(HASH(customer_id, customer_name, segment)) AS customer_key,
            ABS(HASH(city, state, region)) AS country_key,
            ABS(HASH(product_id, product_name, sub_category)) AS product_key,
            SALES,
            QUANTITY,
            DISCOUNT,
            PROFIT,
            left(_FIVETRAN_SYNCED, 16) as _FIVETRAN_SYNCED,
            _sf_ingestion_timestamp
        from
            GLOBAL_SUPERSTORE.bronze_superstore.b_order
        where
            row_id is not null
    );

    /* 
    
    Calender table Silver and insert data query 
    
    */

    
    CREATE    OR REPLACE TABLE global_superstore.Silver_superstore.S_Calender (
        Calender_SK NUMBER AUTOINCREMENT START = 1 INCREMENT = 1,
        CALENDAR_DATE DATE primary key not null,
        YEAR NUMBER,
        QUARTER STRING,
        MONTH STRING,
        WEEKNUMBER STRING,
        DAYNAME STRING,
        DAY NUMBER,
        INGESTION_TIMESTAMP TIMESTAMP_LTZ DEFAULT CURRENT_TIMESTAMP
    )
    --CLUSTER BY (ORDER_DATE)
    DATA_RETENTION_TIME_IN_DAYS = 3 COMMENT = 'Calender table with between MIN(order_date) AS min_date and  MAX(ship_date) + 1 AS max_date';;

    
    /* 
    
    bulk insert data query 
    
    */

    
Insert into
    global_superstore.Silver_superstore.S_Calender (
        CALENDAR_DATE,
        YEAR,
        QUARTER,
        MONTH,
        WEEKNUMBER,
        DAYNAME,
        DAY
    ) 
    (
        WITH date_range AS (
            SELECT
                MIN(order_date) AS min_date,
                MAX(ship_date) + 1 AS max_date
            FROM
                global_superstore.bronze_superstore.b_order
        )
        SELECT
            DATEADD(DAY, SEQ4(), min_date) AS calendar_date,
            YEAR(DATEADD(DAY, SEQ4(), min_date)) AS year,
            concat('Q', QUARTER(DATEADD(DAY, SEQ4(), min_date))) as quarter,
            MONTHNAME(DATEADD(DAY, SEQ4(), min_date)) as month,
            concat('W', week(DATEADD(DAY, SEQ4(), min_date))) as Weeknumber,
            DAYNAME(DATEADD(DAY, SEQ4(), min_date)) as dayname,
            DAY(DATEADD(DAY, SEQ4(), min_date)) as Day
        FROM
            date_range,
            TABLE(GENERATOR(ROWCOUNT => 2000))
        WHERE
            DATEADD(DAY, SEQ4(), min_date) <= max_date
    );

CREATE OR REPLACE PROCEDURE global_superstore.silver_superstore.S_order_procedure()
RETURNS VARCHAR
LANGUAGE SQL
comment = 'S_Order table Procedure '
EXECUTE AS OWNER

AS
$$
BEGIN

/* 
 insert data query 
*/

Insert into  global_superstore.Silver_superstore.S_ORDER (
        ROW_ID,
        ORDER_ID,
        ORDER_DATE,
        SHIP_DATE,
        SHIP_MODE,
        CUSTOMER_KEY,
        COUNTRY_KEY,
        PRODUCT_KEY,
        SALES,
        QUANTITY,
        DISCOUNT,
        PROFIT,
        _FIVETRAN_SYNCED,
        _SF_INGESTION_TIMESTAMP
    ) 
    (
        select 
            distinct ROW_ID,
            ORDER_ID,
            ORDER_DATE,
            SHIP_DATE,
            SHIP_MODE,
            ABS(HASH(customer_id, customer_name, segment)) AS customer_key,
            ABS(HASH(city, state, region)) AS country_key,
            ABS(HASH(product_id, product_name, sub_category)) AS product_key,
            SALES,
            QUANTITY,
            DISCOUNT,
            PROFIT,
            left(_FIVETRAN_SYNCED, 16) as _FIVETRAN_SYNCED,
            _sf_ingestion_timestamp
        from
            GLOBAL_SUPERSTORE.bronze_superstore.b_order
        where
            row_id is not null 
            and  concat (ORDER_ID,'-', ORDER_DATE) not in (select distinct concat (ORDER_ID,'-', ORDER_DATE) from global_superstore.silver_superstore.b_order)
    );



  -- return message the record with date parameter 
  
  RETURN 'Insert and update the table S_Order of Bronze layer table';
END;
$$;
