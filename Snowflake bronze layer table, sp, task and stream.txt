CREATE OR REPLACE TABLE global_superstore.Bronze_superstore.B_order (
    ORDER_SK NUMBER AUTOINCREMENT START = 1000 INCREMENT = 1,
    ROW_ID STRING,
    PRODUCT_ID STRING,
    POSTAL_CODE STRING,
    CATEGORY STRING,
    DISCOUNT STRING,
    ORDER_ID STRING NOT NULL,
    PRODUCT_NAME STRING,
    SHIP_DATE DATE,
    QUANTITY NUMBER,
    CITY STRING,
    SUB_CATEGORY STRING,
    SHIPPING_COST NUMBER,
    SEGMENT STRING,
    SALES NUMBER,

    STATE STRING,

    SHIP_MODE STRING,

/* 
 Order table procedure 
*/

select * from global_superstore.bronze_superstore.b_order;


CREATE OR REPLACE PROCEDURE global_superstore.bronze_superstore.B_order_procedure()
RETURNS VARCHAR
LANGUAGE SQL
comment = 'B_Order table Procedure '
EXECUTE AS OWNER

AS
$$
BEGIN

/* 
 insert data query 
*/


insert into    global_superstore.Bronze_superstore.B_order 
(
        _FIVETRAN_SYNCED,   ROW_ID,             PRODUCT_ID,
        POSTAL_CODE,        CATEGORY,           DISCOUNT,
        ORDER_ID,           PRODUCT_NAME,       SHIP_DATE,
        QUANTITY,           CITY,               SUB_CATEGORY,
        SHIPPING_COST,      SEGMENT,            SALES,
        CUSTOMER_ID,        STATE,              ORDER_PRIORITY,
        SHIP_MODE,          COUNTRY,            REGION,
        ORDER_DATE,         PROFIT,             CUSTOMER_NAME
    ) 
        (
        select
            distinct _FIVETRAN_SYNCED,
            ROW_ID,
            PRODUCT_ID,
            POSTAL_CODE,
            CATEGORY,
            DISCOUNT,
            ORDER_ID,
            PRODUCT_NAME,
            SHIP_DATE,
            QUANTITY,
            CITY,
            SUB_CATEGORY,
            SHIPPING_COST,
            SEGMENT,
            SALES,
            CUSTOMER_ID,
            STATE,
            ORDER_PRIORITY,
            SHIP_MODE,
            COUNTRY,
            REGION,
            ORDER_DATE,
            PROFIT,
            CUSTOMER_NAME
        from    global_superstore.google_sheet.orders_range
        where  ORDER_ID not in (select distinct ORDER_ID from global_superstore.google_sheet.orders_range)
    );


  -- return message the record with date parameter 
  
  RETURN 'Insert and update the table B_Order of Bronze layer table';
END;
$$;

call global_superstore.bronze_superstore.B_order_procedure();

wrapper procedure

CREATE OR REPLACE PROCEDURE global_superstore.bronze_superstore.B_wrapper_procedure()
RETURNS VARCHAR
LANGUAGE SQL
comment = 'B_Wrapper_procedure to wrap process of B_order, B_People, B_Return '
EXECUTE AS OWNER

AS
$$
BEGIN

call global_superstore.bronze_superstore.B_order_procedure();

call global_superstore.bronze_superstore.B_return_procedure();

  RETURN 'B_Wrapper_procedure to wrap process of B_order, B_People, B_Return ';
END;
$$;

call global_superstore.bronze_superstore.B_wrapper_procedure();
